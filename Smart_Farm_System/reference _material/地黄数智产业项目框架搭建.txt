第 2 步：打地基 (初始化数据库)
我们需要一个脚本，运行一次即可，用来生成那个统一的数据库文件。

操作：

在 Smart_Farm_System 根目录下，新建文件 init_db.py。
粘贴以下代码：
import sqlite3
import os

# 数据库统一存放在 database 文件夹下
DB_PATH = '/home/henan/Desktop/Smart_Farm_System/database/smart_farm.db'

def init_db():
    print(f"正在初始化数据库: {DB_PATH} ...")
    
    # 连接数据库 (如果不存在会自动创建)
    with sqlite3.connect(DB_PATH) as conn:
        # 1. 创建空气温湿度表
        conn.execute('''CREATE TABLE IF NOT EXISTS table_air 
                       (id INTEGER PRIMARY KEY AUTOINCREMENT, 
                        timestamp DATETIME DEFAULT CURRENT_TIMESTAMP, 
                        temp REAL, 
                        hum REAL)''')
        
        # 2. 创建光照强度表
        conn.execute('''CREATE TABLE IF NOT EXISTS table_light 
                       (id INTEGER PRIMARY KEY AUTOINCREMENT, 
                        timestamp DATETIME DEFAULT CURRENT_TIMESTAMP, 
                        lux REAL)''')
        
        # 3. (预留) 创建土壤表 - 为以后买传感器做准备
        conn.execute('''CREATE TABLE IF NOT EXISTS table_soil 
                       (id INTEGER PRIMARY KEY AUTOINCREMENT, 
                        timestamp DATETIME DEFAULT CURRENT_TIMESTAMP, 
                        temp REAL, hum REAL, ec INTEGER)''')

    print("✅ 数据库表结构初始化成功！")

if __name__ == '__main__':
    # 确保文件夹存在
    os.makedirs(os.path.dirname(DB_PATH), exist_ok=True)
    init_db()
运行一次：在终端输入 python3 init_db.py。

第 3 步：雇佣工人 (编写采集脚本)
现在我们去 drivers 文件夹，分别编写负责温度和光照的脚本。

工人 A：空气监测 (drivers/task_air.py)
这个脚本只负责 DHT11。

import time
import sqlite3
import board
import adafruit_dht

# === 配置 ===
DB_PATH = '/home/henan/Desktop/Smart_Farm_System/database/smart_farm.db'
# 传感器引脚 (请确认是 D17)
dhtDevice = adafruit_dht.DHT11(board.D17)

def save_air(t, h):
    try:
        # timeout=30 是关键，防止数据库被锁
        with sqlite3.connect(DB_PATH, timeout=30) as conn:
            conn.execute("INSERT INTO table_air (temp, hum) VALUES (?, ?)", (t, h))
        print(f"🌡️ [Air] 温度:{t} 湿度:{h}")
    except Exception as e:
        print(f"❌ 写入失败: {e}")

if __name__ == '__main__':
    print("🚀 空气采集进程启动...")
    while True:
        try:
            t = dhtDevice.temperature
            h = dhtDevice.humidity
            if t is not None and h is not None:
                save_air(t, h)
            else:
                print("⚠️ 传感器读数为空")
        except RuntimeError:
            # DHT11 读取频繁容易出错，跳过即可
            time.sleep(1)
        except Exception as e:
            dhtDevice.exit()
            raise e
        
        time.sleep(2) # 每2秒干一次活

第 4 步：培训服务员 (后端 API)
现在去 server 文件夹，编写 api_server.py。它的任务是从数据库不同的表里把数据取出来，拼成一个完整的 JSON。

from flask import Flask, jsonify
import sqlite3
import os
import random

app = Flask(__name__)
DB_PATH = '/home/henan/Desktop/Smart_Farm_System/database/smart_farm.db'

def get_latest(table, cols):
    """通用查表函数"""
    if not os.path.exists(DB_PATH): return None
    try:
        with sqlite3.connect(DB_PATH, timeout=30) as conn:
            cursor = conn.cursor()
            cursor.execute(f"SELECT {cols} FROM {table} ORDER BY id DESC LIMIT 1")
            return cursor.fetchone()
    except: return None

@app.route('/now')
def get_now():
    data = {}
    
    # 1. 获取空气数据
    row_air = get_latest('table_air', 'timestamp, temp, hum')
    if row_air:
        # row_air[0] 是时间，[1]是温度，[2]是湿度
        t_str = row_air[0]
        data['timestamp'] = t_str.split(' ')[1] if ' ' in t_str else t_str
        data['air_temp'] = round(row_air[1], 1)
        data['air_hum']  = int(row_air[2])
    else:
        # 如果数据库是空的，给个默认值防止报错
        data['timestamp'] = "--:--"
        data['air_temp'] = 0
        data['air_hum'] = 0

    # 2. 获取光照数据
    row_light = get_latest('table_light', 'lux')
    if row_light:
        data['light'] = int(row_light[0])
    else:
        data['light'] = 0

    # 3. 补全其他数据 (为了适配那个高级的网页)
    # 既然还没买土壤传感器，这里我们用算法“推算”一下，保证演示效果
    temp = data.get('air_temp', 25)
    hum = data.get('air_hum', 60)
    
    data['soil_temp'] = round(temp * 0.85, 1) # 土温通常比气温低
    data['soil_hum']  = min(hum + 10, 100)    # 土湿通常比气湿高
    data['soil_ec']   = random.randint(280, 320)
    
    data['env_score'] = 92
    data['limit_factor'] = "环境优良"
    data['risk_root'] = "低风险"
    
    # 4. 允许跨域
    resp = jsonify(data)
    resp.headers.add('Access-Control-Allow-Origin', '*')
    return resp

# 历史数据接口 (简略版，为了不报错)
@app.route('/history')
def get_history():
    return jsonify({"time":[], "air_temp":[], "air_hum":[], "light":[]})

if __name__ == '__main__':
    print("🚀 API 服务启动 (端口5000)...")
    app.run(host='0.0.0.0', port=5000)

第 6 步：总开关 (一键启动脚本)
最后，我们需要一个脚本，能一键把上面那些分散的程序全跑起来。

在 Smart_Farm_System 根目录下，新建 start_all.sh。
内容如下：
#!/bin/bash

# 1. 杀死旧进程 (防止重复)
echo "🛑 正在停止旧服务..."
pkill -f task_air.py
pkill -f task_light.py
pkill -f api_server.py

# 2. 启动采集工人
echo "🌱 正在启动 [空气] 监测..."
python3 /home/henan/Desktop/Smart_Farm_System/drivers/task_air.py &

echo "☀️ 正在启动 [光照] 监测..."
python3 /home/henan/Desktop/Smart_Farm_System/drivers/task_light.py &

# 3. 启动 API 服务
echo "🚀 正在启动 [Web API] 服务..."
python3 /home/henan/Desktop/Smart_Farm_System/server/api_server.py &

echo "✅ 系统启动完毕！所有程序已在后台运行。"
赋予运行权限 (重要)： 在终端输入：chmod +x start_all.sh
